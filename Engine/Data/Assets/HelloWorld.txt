using System;
using FiestaEngine;

public class HelloWorld : Festejo
{
		public float speed;	

        private float angle_x;
       // private float angle_y;
        public float rotation_speed_x;
       // public float rotation_speed_y;

        bool Start()
        {
            return true;
        }

		bool Update()
	    	{

			if(FiestaInput.GetKeyRepeat(FiestaInput.FiestaKeyCode.W))
			{
				
                trans.local_position =   trans.local_position + (trans.forward.Normalized * speed * Time.delta_time);
      
			}

            if(FiestaInput.GetKeyRepeat(FiestaInput.FiestaKeyCode.S))
			{
				
                trans.local_position =   trans.local_position + (trans.forward.Normalized * -speed * Time.delta_time);
      
			}

            if(FiestaInput.GetKeyRepeat(FiestaInput.FiestaKeyCode.A))
			{
				
                trans.local_position =   trans.local_position + (trans.right.Normalized * speed * Time.delta_time);
      
			}

            if(FiestaInput.GetKeyRepeat(FiestaInput.FiestaKeyCode.D))
			{
				
                trans.local_position =   trans.local_position + (trans.right.Normalized * -speed * Time.delta_time);
      
			}
			
          angle_x = angle_x + (rotation_speed_x * FiestaInput.GetMouseXAxis());
         // angle_y = angle_y + (rotation_speed_y * FiestaInput.GetMouseYAxis() * Time.delta_time);

          trans.euler_angles = new FiestaVector3(0.0f, angle_x, 0.0f);
         // trans.euler_angles = new FiestaVector3(angle_y, 0.0f, 0.0f);            

			return true;		
		}
	
}
